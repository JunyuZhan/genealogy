// Simple wrapper for fetch API
// In a real application, this would handle base URLs, auth tokens, error handling, etc.
 
const BASE_URL = '/api';
 
export async function get<T>(endpoint: string): Promise<T> {
  const response = await fetch(`${BASE_URL}${endpoint}`);
  if (!response.ok) {
    throw new Error(`API Error: ${response.statusText}`);
  }
  return response.json();
}
 
export async function post<T>(endpoint: string, data: any): Promise<T> {
  // For this mock backend (static files), we can't really POST.
  // We'll simulate a successful response.
  console.log(`[Mock POST] ${endpoint}`, data);
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(data as T);
    }, 500);
  });
}
